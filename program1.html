Kruskal's Algorithm: C Program

The following C program finds the Minimum Cost Spanning Tree using Kruskal's algorithm:

#include <stdio.h>

#define INFINITY 999
#define MAX 10

struct EDGE {
    int x, y, wt;       
} e[MAX];

int parent[MAX], cost[MAX][MAX], t[MAX][2];
int nedges, eno;

int get_parent(int v) {
    while (parent[v] > 0)
        v = parent[v];
    return v;
}

void sort_edges() {
    struct EDGE temp;
    for (int i = 1; i < nedges; i++) {
        for (int j = i + 1; j <= nedges; j++) {
            if (e[i].wt > e[j].wt) {
                temp = e[i];
                e[i] = e[j];
                e[j] = temp;
            }
        }
    }
}

void kruskal(int n) {
    int i, j = 1, k = 1, sum = 0;
    while (j < n) {
        int u = e[k].x;
        int v = e[k].y;
        int pu = get_parent(u);
        int pv = get_parent(v);
        if (pu != pv) {
            t[j][0] = u;
            t[j][1] = v;
            parent[pu] = pv;
            sum += e[k].wt;
            j++;
        }
        k++;
    }
    printf("\nCost of the spanning tree is: %d\n", sum);
    printf("Edges of the spanning tree are:\n");
    for (i = 1; i < n; i++)
        printf("%d -> %d\n", t[i][0], t[i][1]);
}

int main() {
    int n, i, j;
    printf("Enter the number of vertices: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++)
        parent[i] = 0;

    eno = 1;
    printf("Enter the cost adjacency matrix:\n");
    for (i = 1; i <= n; i++) {
        for (j = 1; j <= n; j++) {
            scanf("%d", &cost[i][j]);
            if (i == j || cost[i][j] == INFINITY) continue;
            e[eno].x = i;
            e[eno].y = j;
            e[eno].wt = cost[i][j];
            eno++;
            nedges++;
        }
    }

    sort_edges();
    kruskal(n);
    return 0;
}
